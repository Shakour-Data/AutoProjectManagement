        <CardHeader>

        </CardHeader>
        <CardContent>
          <div>Loading storage status...</div>
        </CardContent>
      </Card>
    );
  }

  const usagePercentage = storageInfo ? (storageInfo.used / storageInfo.total) * 100 : 0;

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          {getModeIcon()}
          Storage Status
          <Badge variant="outline" className="ml-auto">
            {mode}
          </Badge>
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        {/* Storage Usage */}
        <div className="space-y-2">
          <div className="flex items-center justify-between text-sm">
            <span>Storage Usage</span>
            <span>{usagePercentage.toFixed(1)}%</span>
          </div>
          <Progress value={usagePercentage} className="h-2" />
          <div className="flex items-center justify-between text-xs text-muted-foreground">
            <span>{formatBytes(storageInfo?.used || 0)} used</span>
            <span>{formatBytes(storageInfo?.total || 0)} total</span>
          </div>
        </div>

        {/* Storage Details */}
        <div className="grid grid-cols-2 gap-4 text-sm">
          {mode === 'file-based' && (
            <div className="space-y-1">
              <div className="flex items-center gap-2">
                <FileText className="h-4 w-4" />
                <span>Files</span>
              </div>
              <div className="text-2xl font-bold">{storageInfo?.fileCount || 0}</div>
            </div>
          )}
          
          {mode === 'database' && (
            <div className="space-y-1">
              <div className="flex items-center gap-2">
                <Database className="h-4 w-4" />
                <span>Records</span>
              </div>
              <div className="text-2xl font-bold">{storageInfo?.recordCount || 0}</div>
            </div>
          )}
          
          {mode === 'hybrid' && (
            <>
              <div className="space-y-1">
                <div className="flex items-center gap-2">
                  <FileText className="h-4 w-4" />
                  <span>Files</span>
                </div>
                <div className="text-2xl font-bold">{storageInfo?.fileCount || 0}</div>
              </div>
              <div className="space-y-1">
                <div className="flex items-center gap-2">
                  <Database className="h-4 w-4" />
                  <span>Records</span>
                </div>
                <div className="text-2xl font-bold">{storageInfo?.recordCount || 0}</div>
              </div>
            </>
          )}
        </div>

        {/* System Health */}
        <div className="flex items-center justify-between">
          <span className="text-sm">System Health</span>
          <div className="flex items-center gap-1">
            {status?.healthy ? (
              <>
                <CheckCircle className="h-4 w-4 text-green-500" />
                <span className="text-green-500">Healthy</span>
              </>
            ) : (
              <>
                <AlertCircle className="h-4 w-4 text-red-500" />
                <span className="text-red-500">Issues Detected</span>
              </>
            )}
          </div>
        </div>

        {/* Actions */}
        <div className="flex gap-2">
          <Button variant="outline" size="sm" onClick={fetchStorageStatus}>
            Refresh
          </Button>
          <Button variant="outline" size="sm">
            <HardDrive className="h-4 w-4 mr-2" />
            Manage Storage
          </Button>
        </div>
      </CardContent>
    </Card>
  );
}

function formatBytes(bytes: number): string {
  if (bytes === 0) return '0 Bytes';
  const k = 1024;
  const sizes = ['Bytes', 'KB', 'MB', 'GB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
}
